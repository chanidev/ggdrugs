## 1. 개요
본 문서는 Alle 서비스의 관계형 데이터 모델을 기술하는 테이블 정의서다. 데이터베이스는 PostgreSQL 15(PostGIS 확장)을 사용하며, 총 **43개 모델(테이블)**로 구성된다. ORM 은 Prisma 를 채택했고, 벡터 검색은 Qdrant v1.13(스키마 외부)이 담당한다.
물리 명명: Prisma 모델은 PascalCase, 필드는 camelCase(@map으로 snake_case 컬럼에 매핑), 테이블명은 snake_case(@@map).
확장 모듈: postgis(공간 데이터), pg_trgm(유사도 검색), unaccent(발음 부호 제거), citext(대소문자 무시 텍스트).
enum 정책: PostgreSQL 네이티브 enum 을 사용하지 않고, VARCHAR + CHECK 제약 또는 마스터 테이블로 도메인을 관리한다(마이그레이션 변경 비용 최소화 목적).
상태 머신(이벤트): pending → revision_requested → pending(재제출) ; pending → approved → ended(종료일 도래, phase 로 이관) ; pending → rejected.
본 문서의 컬럼 정의는 실제 데이터베이스 스키마(Prisma ORM 기반)를 1차 근거로 한다.

## 2. 표기 규약 및 Prisma → SQL 매핑
명세표는 다음 컬럼을 사용한다.
Prisma 타입 → PostgreSQL 타입 매핑:
CHECK 제약은 Prisma 가 직접 표현하지 못하므로 초기 마이그레이션 SQL 에서 수동 보강한다. updated_at 자동 갱신 트리거(fn_set_updated_at) 역시 마이그레이션 SQL 에서 생성한다.

## 3. 핵심 테이블 명세
### 3.1 regions (지역)
시·도/시·군·구/동 3계층 행정구역. 거리 정렬의 anchor 가 되는 구청 중심좌표(center_lat/lng)는 시·군·구 단위만 채운다.
CREATE TABLE regions (
  region_id    BIGSERIAL     PRIMARY KEY,
  sido_name    VARCHAR(30)   NOT NULL,
  sigungu_name VARCHAR(30),
  dong_name    VARCHAR(30),
  full_address VARCHAR(100)  NOT NULL,
  center_lat   NUMERIC(10,7),
  center_lng   NUMERIC(10,7),
  created_at   TIMESTAMPTZ   NOT NULL DEFAULT now()
);
CREATE INDEX idx_regions_sido    ON regions (sido_name);
CREATE INDEX idx_regions_sigungu ON regions (sido_name, sigungu_name);

### 3.2 users (사용자)
"1계정 = 복수 역할 토글" 모델의 중심 테이블. 별도 계정 생성 없이 일반 사용자 계정에 업로더 역할을 추가한다. active_role이 현재 활성 역할을 표현한다(admin 여부는 admin_profiles로 식별).
CREATE TABLE users (
  user_id             BIGSERIAL    PRIMARY KEY,
  social_uid          VARCHAR(255) NOT NULL,
  auth_provider       VARCHAR(20)  NOT NULL,
  nickname            VARCHAR(50)  NOT NULL,
  gender              CHAR(1),
  date_of_birth       DATE,
  region_id           BIGINT       REFERENCES regions(region_id),
  active_role         VARCHAR(20)  NOT NULL DEFAULT 'user',
  is_notification_on  BOOLEAN      NOT NULL DEFAULT false,
  is_deleted          BOOLEAN      NOT NULL DEFAULT false,
  last_logged_in_at   TIMESTAMPTZ,
  created_at          TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at          TIMESTAMPTZ  NOT NULL DEFAULT now(),
  deleted_at          TIMESTAMPTZ,
  sanction_status     VARCHAR(20)  NOT NULL DEFAULT 'none',
  sanction_expires_at TIMESTAMPTZ,
  sanction_reason     TEXT,
  CONSTRAINT uq_users_social UNIQUE (auth_provider, social_uid),
  CONSTRAINT chk_users_active_role     CHECK (active_role IN ('user','uploader')),
  CONSTRAINT chk_users_gender          CHECK (gender IN ('M','F')),
  CONSTRAINT chk_users_sanction_status CHECK (sanction_status IN ('none','warned','suspended'))
);
CREATE INDEX idx_users_provider_uid ON users (auth_provider, social_uid);
CREATE INDEX idx_users_region       ON users (region_id);
CREATE INDEX idx_users_created      ON users (created_at);

### 3.3 uploader_profiles (업로더 프로필)
users 와 1:1. 업로더 승급 시 생성되는 PII 확장 테이블. 기관 업로더의 사업자등록번호(business_registration_number)와 개인 업로더의 본인인증 CI(ci_hash)는 XOR 관계(둘 중 하나만 채움).
CREATE TABLE uploader_profiles (
  uploader_id                  BIGSERIAL    PRIMARY KEY,
  user_id                      BIGINT       NOT NULL UNIQUE REFERENCES users(user_id),
  organization_name            VARCHAR(100) NOT NULL,
  contact_phone                VARCHAR(20)  NOT NULL,
  contact_email                VARCHAR(255) NOT NULL,
  real_name                    VARCHAR(50)  NOT NULL DEFAULT '',
  business_registration_number CHAR(10),
  ci_hash                      CHAR(88),
  approval_status              VARCHAR(20)  NOT NULL DEFAULT 'pending',
  approved_at                  TIMESTAMPTZ,
  created_at                   TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at                   TIMESTAMPTZ  NOT NULL DEFAULT now(),
  CONSTRAINT chk_uploader_status CHECK (approval_status IN ('pending','approved','revision_requested','rejected'))
);
CREATE INDEX idx_uploader_status ON uploader_profiles (approval_status);

### 3.4 uploader_documents (업로더 승급 서류)
업로더 승급 심사용 첨부 서류. 이벤트 심사용 approval_documents 와 분리한다. 업로더 프로필 삭제 시 CASCADE.
CREATE TABLE uploader_documents (
  document_id       BIGSERIAL    PRIMARY KEY,
  uploader_id       BIGINT       NOT NULL REFERENCES uploader_profiles(uploader_id) ON DELETE CASCADE,
  file_path         VARCHAR(500) NOT NULL,
  original_filename VARCHAR(255) NOT NULL,
  mime_type         VARCHAR(30)  NOT NULL,
  file_size_bytes   INTEGER      NOT NULL,
  created_at        TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_uploader_docs_uploader ON uploader_documents (uploader_id);

### 3.5 admin_profiles (관리자 프로필)
users 와 1:1. 관리자 권한 범위(scope)를 표현한다.
CREATE TABLE admin_profiles (
  admin_id   BIGSERIAL    PRIMARY KEY,
  user_id    BIGINT       NOT NULL UNIQUE REFERENCES users(user_id),
  department VARCHAR(100),
  scope      VARCHAR(30)  NOT NULL DEFAULT 'full',
  is_active  BOOLEAN      NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ  NOT NULL DEFAULT now(),
  CONSTRAINT chk_admin_scope CHECK (scope IN ('full','content_only','uploader_review_only'))
);
CREATE INDEX idx_admin_active ON admin_profiles (is_active);

### 3.6 event_categories (이벤트 종류)
이벤트 종류 마스터. category_code는 {festival, expo, symposium, conference} 도메인.
CREATE TABLE event_categories (
  category_id   BIGSERIAL   PRIMARY KEY,
  category_code VARCHAR(30) NOT NULL UNIQUE,
  display_name  VARCHAR(50) NOT NULL,
  sort_order    SMALLINT    NOT NULL DEFAULT 0,
  is_active     BOOLEAN     NOT NULL DEFAULT true,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT chk_event_category_code CHECK (category_code IN ('festival','expo','symposium','conference'))
);

### 3.7 events (이벤트)
모든 행사의 상위 개념(절대 festival 로 대체하지 않는다). 크롤링 수집 이벤트는 uploader_id가 NULL. location_geom은 PostGIS geometry(Point,4326)으로 raw query 전용(Prisma client 에서 제외). LLM 요약(ai_summary)은 gpt-4o-mini 가 생성하며, description 변경 시 트리거가 ai_summary를 무효화한다. 종료일 도래 시 상태는 phase = 'ended'로 이관된다.
CREATE TABLE events (
  event_id                     BIGSERIAL    PRIMARY KEY,
  uploader_id                  BIGINT       REFERENCES uploader_profiles(uploader_id),
  category_id                  BIGINT       NOT NULL REFERENCES event_categories(category_id),
  region_id                    BIGINT       NOT NULL REFERENCES regions(region_id),
  source_type                  VARCHAR(20)  NOT NULL,
  crawl_origin                 VARCHAR(50),
  external_source_id           VARCHAR(100),
  title                        VARCHAR(200) NOT NULL,
  description                  TEXT,
  ai_summary                   TEXT,
  ai_summary_at                TIMESTAMPTZ,
  description_hash             CHAR(32),
  address_detail               VARCHAR(255),
  location_geom                geometry(Point, 4326),
  start_date                   DATE         NOT NULL,
  end_date                     DATE         NOT NULL,
  operating_hours              VARCHAR(100),
  target_audience              VARCHAR(100),
  admission_fee                VARCHAR(100),
  expected_companion_primary   VARCHAR(20),
  expected_companion_secondary VARCHAR(20),
  poster_image_url             VARCHAR(500),
  approval_status              VARCHAR(20)  NOT NULL DEFAULT 'pending',
  phase                        VARCHAR(20)  NOT NULL DEFAULT 'upcoming',
  bookmark_count               INTEGER      NOT NULL DEFAULT 0,
  avg_rating                   NUMERIC(3,2) NOT NULL DEFAULT 0,
  review_count                 INTEGER      NOT NULL DEFAULT 0,
  is_deleted                   BOOLEAN      NOT NULL DEFAULT false,
  approved_at                  TIMESTAMPTZ,
  created_at                   TIMESTAMPTZ  NOT NULL DEFAULT now(),
  updated_at                   TIMESTAMPTZ  NOT NULL DEFAULT now(),
  deleted_at                   TIMESTAMPTZ,
  CONSTRAINT uq_events_external UNIQUE (crawl_origin, external_source_id),
  CONSTRAINT chk_events_source_type     CHECK (source_type IN ('crawled','uploaded')),
  CONSTRAINT chk_events_approval_status CHECK (approval_status IN ('pending','approved','revision_requested','rejected')),
  CONSTRAINT chk_events_phase           CHECK (phase IN ('upcoming','ongoing','ended'))
);
CREATE INDEX idx_events_filter   ON events (region_id, category_id, start_date, phase);
CREATE INDEX idx_events_phase    ON events (phase, start_date);
CREATE INDEX idx_events_approval ON events (approval_status, created_at);
CREATE INDEX idx_events_uploader ON events (uploader_id);
CREATE INDEX idx_events_dates    ON events (start_date, end_date);
-- 공간 인덱스(raw query 근접/bbox 용)
-- 마이그레이션 SQL 에서 수동 생성(Prisma @@index 아님 — location_geom 이 Unsupported("geometry(Point,4326)") 라 Prisma 가 인덱스를 표현 못 함)
CREATE INDEX idx_events_geom     ON events USING GIST (location_geom);

### 3.8 event_vibes (이벤트 성향 마스터)
관리자가 부여하는 이벤트 성향(event_vibe) 라벨의 마스터.
CREATE TABLE event_vibes (
  vibe_id    BIGSERIAL   PRIMARY KEY,
  vibe_name  VARCHAR(50) NOT NULL UNIQUE,
  vibe_group VARCHAR(30) NOT NULL,
  is_active  BOOLEAN     NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT chk_event_vibe_group CHECK (vibe_group IN ('mood','activity','theme'))
);

### 3.9 event_vibe_assignments (이벤트 성향 부여)
이벤트 ↔ 성향 N:M. 성향 부여는 관리자(사람)의 결정(assigned_by). LLM 위임 금지.
CREATE TABLE event_vibe_assignments (
  assignment_id BIGSERIAL   PRIMARY KEY,
  event_id      BIGINT      NOT NULL REFERENCES events(event_id),
  vibe_id       BIGINT      NOT NULL REFERENCES event_vibes(vibe_id),
  assigned_by   BIGINT      NOT NULL REFERENCES users(user_id),
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT uq_event_vibe UNIQUE (event_id, vibe_id)
);
CREATE INDEX idx_vibe_assign_event ON event_vibe_assignments (event_id);
CREATE INDEX idx_vibe_assign_vibe  ON event_vibe_assignments (vibe_id);

### 3.10 approval_documents (이벤트 심사 첨부)
이벤트 승인 심사용 첨부 자료(이미지).
CREATE TABLE approval_documents (
  document_id       BIGSERIAL    PRIMARY KEY,
  event_id          BIGINT       NOT NULL REFERENCES events(event_id),
  file_path         VARCHAR(500) NOT NULL,
  original_filename VARCHAR(255) NOT NULL,
  mime_type         VARCHAR(30)  NOT NULL,
  file_size_bytes   INTEGER      NOT NULL,
  created_at        TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_approval_docs_event ON approval_documents (event_id);

### 3.11 approval_logs (승인 이력)
이벤트 심사 액션 이력. 액션은 관리자(사람) 결정.
CREATE TABLE approval_logs (
  log_id     BIGSERIAL   PRIMARY KEY,
  event_id   BIGINT      NOT NULL REFERENCES events(event_id),
  admin_id   BIGINT      NOT NULL REFERENCES users(user_id),
  action     VARCHAR(20) NOT NULL,
  reason     TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT chk_approval_log_action CHECK (action IN ('approved','revision_requested','rejected'))
);
CREATE INDEX idx_approval_logs_event ON approval_logs (event_id, created_at DESC);
CREATE INDEX idx_approval_logs_admin ON approval_logs (admin_id, created_at DESC);

### 3.12 bookmarks (북마크)
사용자 × 이벤트 북마크. 쌍 UNIQUE.
CREATE TABLE bookmarks (
  bookmark_id BIGSERIAL   PRIMARY KEY,
  user_id     BIGINT      NOT NULL REFERENCES users(user_id),
  event_id    BIGINT      NOT NULL REFERENCES events(event_id),
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT uq_bookmark UNIQUE (user_id, event_id)
);
CREATE INDEX idx_bookmarks_user  ON bookmarks (user_id, created_at DESC);
CREATE INDEX idx_bookmarks_event ON bookmarks (event_id);

### 3.13 reviews (리뷰)
이벤트 리뷰. 사용자당 이벤트별 1건(UNIQUE). sentiment는 LLM 분석 보조 결과(감정 분류).
CREATE TABLE reviews (
  review_id  BIGSERIAL   PRIMARY KEY,
  user_id    BIGINT      NOT NULL REFERENCES users(user_id),
  event_id   BIGINT      NOT NULL REFERENCES events(event_id),
  body       TEXT        NOT NULL,
  rating     SMALLINT    NOT NULL,
  sentiment  VARCHAR(10),
  is_deleted BOOLEAN     NOT NULL DEFAULT false,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  deleted_at TIMESTAMPTZ,
  CONSTRAINT uq_review_per_event UNIQUE (user_id, event_id),
  CONSTRAINT chk_reviews_sentiment CHECK (sentiment IN ('positive','negative','neutral'))
);
CREATE INDEX idx_reviews_event     ON reviews (event_id, created_at DESC);
CREATE INDEX idx_reviews_user      ON reviews (user_id, created_at DESC);
CREATE INDEX idx_reviews_sentiment ON reviews (event_id, sentiment);

### 3.14 review_photos (리뷰 사진)
리뷰 첨부 사진(최대 5장). 리뷰 삭제 시 CASCADE.
CREATE TABLE review_photos (
  review_photo_id   BIGSERIAL    PRIMARY KEY,
  review_id         BIGINT       NOT NULL REFERENCES reviews(review_id) ON DELETE CASCADE,
  file_path         VARCHAR(500) NOT NULL,
  original_filename VARCHAR(255) NOT NULL,
  mime_type         VARCHAR(30)  NOT NULL,
  file_size_bytes   INTEGER      NOT NULL,
  sort_order        SMALLINT     NOT NULL DEFAULT 0,
  created_at        TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_review_photos_review ON review_photos (review_id, sort_order);

### 3.15 notifications (알림)
예약 발송·조건 기반 알림. related_entity_id/type은 알림 클릭 라우팅에 사용.
CREATE TABLE notifications (
  notification_id     BIGSERIAL    PRIMARY KEY,
  user_id             BIGINT       NOT NULL REFERENCES users(user_id),
  event_id            BIGINT       REFERENCES events(event_id),
  title               VARCHAR(100) NOT NULL,
  message             TEXT         NOT NULL,
  scheduled_at        TIMESTAMPTZ  NOT NULL,
  is_sent             BOOLEAN      NOT NULL DEFAULT false,
  sent_at             TIMESTAMPTZ,
  read_at             TIMESTAMPTZ,
  notification_type   VARCHAR(30),
  related_entity_id   BIGINT,
  related_entity_type VARCHAR(30),
  created_at          TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_notif_pending ON notifications (scheduled_at);
CREATE INDEX idx_notif_user    ON notifications (user_id, created_at DESC);

### 3.16 search_logs (검색 로그)
필터 검색·채팅 검색 행동 로그. 파라미터는 JSONB 로 저장(파티셔닝은 Phase 3+ 연기).
CREATE TABLE search_logs (
  log_id        BIGSERIAL   PRIMARY KEY,
  user_id       BIGINT      NOT NULL REFERENCES users(user_id),
  search_type   VARCHAR(10) NOT NULL,
  search_params JSONB       NOT NULL DEFAULT '{}',
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT chk_search_type CHECK (search_type IN ('filter','chat'))
);
CREATE INDEX idx_search_logs_user ON search_logs (user_id, created_at DESC);

### 3.17 chat_sessions / chat_messages (LLM 검색 세션·메시지)
자연어 이벤트 검색을 위한 LLM 대화 세션·메시지. 동행 매칭의 chat_rooms/chat_room_messages 와 완전히 별개(JOIN 금지). LLM 은 추천·검색·요약에만 사용.
chat_sessions
CREATE TABLE chat_sessions (
  session_id BIGSERIAL   PRIMARY KEY,
  user_id    BIGINT      NOT NULL REFERENCES users(user_id),
  is_active  BOOLEAN     NOT NULL DEFAULT true,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  ended_at   TIMESTAMPTZ
);
CREATE INDEX idx_chat_sessions_user ON chat_sessions (user_id, created_at DESC);
chat_messages
CREATE TABLE chat_messages (
  message_id  BIGSERIAL   PRIMARY KEY,
  session_id  BIGINT      NOT NULL REFERENCES chat_sessions(session_id),
  sender_type VARCHAR(10) NOT NULL,
  body        TEXT        NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT chk_chat_msg_sender CHECK (sender_type IN ('user','assistant'))
);
CREATE INDEX idx_chat_msg_session ON chat_messages (session_id, created_at);

### 3.18 news_articles (외부 뉴스)
외부 뉴스 기사(네이버 뉴스 검색 API · Google News RSS로 수집). original_url은 중복 적재 방지를 위해 UNIQUE. 이벤트 매핑 커버리지는 44%(1,810/4,111).
CREATE TABLE news_articles (
  article_id       BIGSERIAL    PRIMARY KEY,
  source_name      VARCHAR(30)  NOT NULL DEFAULT 'donga',
  author_name      VARCHAR(50),
  article_category VARCHAR(50),
  title            VARCHAR(300) NOT NULL,
  original_url     VARCHAR(500) NOT NULL UNIQUE,
  content_body     TEXT,
  summary          TEXT,
  metadata         JSONB        NOT NULL DEFAULT '{}',
  published_at     TIMESTAMPTZ,
  crawled_at       TIMESTAMPTZ  NOT NULL DEFAULT now(),
  created_at       TIMESTAMPTZ  NOT NULL DEFAULT now()
);
CREATE INDEX idx_articles_published ON news_articles (published_at DESC);
CREATE INDEX idx_articles_source    ON news_articles (source_name, published_at DESC);

### 3.19 event_article_mappings (이벤트↔기사 매핑)
이벤트와 뉴스 기사의 연관 매핑. relevance_score는 0~1 정밀도의 관련도(산출식 (TBD)).
CREATE TABLE event_article_mappings (
  mapping_id      BIGSERIAL    PRIMARY KEY,
  event_id        BIGINT       NOT NULL REFERENCES events(event_id),
  article_id      BIGINT       NOT NULL REFERENCES news_articles(article_id),
  relevance_score NUMERIC(5,4) NOT NULL DEFAULT 0,
  matched_at      TIMESTAMPTZ  NOT NULL DEFAULT now(),
  CONSTRAINT uq_event_article UNIQUE (event_id, article_id)

## 5. enum 도메인 정의
네이티브 enum 을 사용하지 않고 VARCHAR + CHECK 또는 마스터 테이블로 관리한다.
거버넌스 원칙: 이벤트 승인·이벤트 성향 부여·업로더 승급 심사·신고 처리 액션은 모두 관리자(사람)의 결정 영역이다. LLM 은 추천·검색·요약에만 사용하며, 위 관리자 판단을 위임하지 않는다.
